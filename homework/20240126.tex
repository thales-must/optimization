  \section{Big M AND The Two-Phase Method}
  Max $z = 2x_1 + 3x_2$


  s.t.
  \begin{itemize}
    \item $x_1 + 2x_2 \leq 4$
    \item $x_1 + x_2 = 3$
    \item $x_1, x_2 \geq 0$
  \end{itemize}


\begin{equation}
  \begin{aligned}
    &z  &-2x_1 &-3x_2 & &+M\bar{x}_4 &= 0 \\
    & &x_1  &+2x_2  &+x_3  &  &= 4 \\
    & &x_1  &+x_2  &  &+\bar{x}_4 &= 3 \\
  \end{aligned}
\end{equation}

\begin{equation}
  \begin{aligned}
    Z -2x_1 -3x_2 + M \bar{x}_4 &= 0 \\
    -M( x_1 + x_2 + \bar{x}_4 &= 3) \\
    \textbf{new} \quad Z - (M + 2)x_1 - (M+3)x_2 &= -3M \\
  \end{aligned}
\end{equation}
